unit Eq;

concept T : Eq
{
    Equals = function bool(T, T);
    NotEquals = function bool(T, T);
}

instance int : Eq
{
    Equals = IntEquals;
    NotEquals = IntNotEquals;
}

function bool IntEquals(int a, int b)
{
    result = a == b;
}

function bool IntNotEquals(int a, int b)
{
    result = a != b;
}
